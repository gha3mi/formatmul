   pure subroutine mat_vec_rel_ATv_opt(A, v, w, option)
      real(rk), intent(in), contiguous :: A(:,:), v(:)
      real(rk), intent(inout), contiguous :: w(:)
      character(*), intent(in) :: option

      select case (option)
       case ('m1')
         call mv_ATv_1(A, v, w)
       case ('m2')
         call mv_ATv_2(A, v, w)
       case ('m3')
         call mv_ATv_3(A, v, w)
       case ('m4')
         call mv_ATv_4(A, v, w)
       case ('m5')
         call mv_ATv_5(A, v, w)
       case ('m6')
         call mv_ATv_6(A, v, w)
       case ('m7')
         call mv_ATv_7(A, v, w)
       case ('m8')
         call mv_ATv_8(A, v, w)
      end select

   end subroutine mat_vec_rel_ATv_opt
