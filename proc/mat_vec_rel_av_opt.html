   pure subroutine mat_vec_rel_Av_opt(A, v, w, option)
      real(rk), intent(in), contiguous :: A(:,:), v(:)
      real(rk), intent(inout), contiguous :: w(:)
      character(*), intent(in) :: option

      select case (option)
       case ('m1')
         call mv_Av_1(A, v, w)
       case ('m2')
         call mv_Av_2(A, v, w)
       case ('m3')
         call mv_Av_3(A, v, w)
       case ('m4')
         call mv_Av_4(A, v, w)
