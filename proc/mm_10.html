   pure subroutine mm_10(m, n, p, a, b, c)
      integer, intent(in) :: m, n, p
      real(rk), intent(in) :: a(m,n), b(n,p)
      real(rk), intent(out) :: c(m,p)
      integer :: i, k
      c = 0.0_rk
      do concurrent (i = 1: p)
         do k = 1, m
            c(k,i) = dot_product(a(k,:), b(:,i))
         end do
      end do
   end subroutine mm_10
