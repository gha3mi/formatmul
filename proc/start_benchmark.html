   subroutine start_benchmark(ti,im,msg)
      type(timer), intent(out) :: ti
      integer,     intent(in)  :: im
      character(*),intent(in)  :: msg
      sync all
      if (this_image() == im) then
         print'(a)', msg
         print'(a,i2)', ' Image        :',im
         call ti%timer_start()
      end if
   end subroutine start_benchmark
